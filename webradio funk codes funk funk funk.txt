#include <Arduino.h>
#include <WiFi.h>
#include <SPI.h>
#include <U8g2lib.h>
#include <VS1053Driver.h>
#include <secrets.h>

// ================= DISPLAY =================
U8G2_SH1106_128X64_NONAME_F_4W_HW_SPI u8g2(
  U8G2_R0,
  25,
  26,
  27
);

// ================= VS1053 =================
#define VS1053_CS    5
#define VS1053_DCS   16
#define VS1053_DREQ  4
#define VS1053_RST   17

SPIClass VSPIbus(VSPI);
VS1053 player(VS1053_CS, VS1053_DCS, VS1053_DREQ, VS1053_RST, VSPIbus);

// ================= ENCODER =================
#define ENC_CLK 35
#define ENC_DT  34
#define ENC_SW  32

int volume = 70;
int lastCLK;

// ================= WIFI =================
WiFiClient client;
const char* ssid = WIFI_SSID;
const char* password = WIFI_PASS;

const char* host = "ic.radiomonster.fm";
const char* path = "/evergreens.ultra";
const int httpPort = 80;

// größerer Buffer
uint8_t mp3buff[128];

bool headerSkipped = false;

// =========================================

void drawScreen()
{
  u8g2.clearBuffer();

  u8g2.setFont(u8g2_font_logisoso18_tf);
  u8g2.drawStr(0, 20, "WEBRADIO");

  char volText[20];
  sprintf(volText, "%d %%", volume);

  u8g2.setFont(u8g2_font_logisoso22_tf);
  u8g2.drawStr(0, 63, volText);

  u8g2.sendBuffer();
}

// =========================================

void connectToStream()
{
  Serial.println("Verbinde Stream");

  client.stop();
  headerSkipped = false;

  if (!client.connect(host, httpPort))
  {
    Serial.println("Stream Fehler");
    return;
  }

  client.print(String("GET ") + path + " HTTP/1.1\r\n" +
               "Host: " + host + "\r\n" +
               "Connection: close\r\n\r\n");

  Serial.println("Stream verbunden");
}

// =========================================

void setup()
{
  Serial.begin(115200);

  pinMode(ENC_CLK, INPUT_PULLUP);
  pinMode(ENC_DT, INPUT_PULLUP);
  pinMode(ENC_SW, INPUT_PULLUP);

  lastCLK = digitalRead(ENC_CLK);

  // Display SPI
  SPI.begin(14, -1, 13, 25);
  u8g2.begin();

  // VS1053 SPI
  VSPIbus.begin(18, 19, 23, 5);
  VSPIbus.setFrequency(1000000);

  if (!player.begin())
  {
    Serial.println("VS1053 Fehler");
    while(true);
  }

  player.switchToMp3Mode();
  player.setVolume(volume);

  Serial.println("VS1053 OK");

  WiFi.begin(ssid, password);

  Serial.print("Verbinde WLAN");

  while (WiFi.status() != WL_CONNECTED)
  {
    delay(500);
    Serial.print(".");
  }

  Serial.println();
  Serial.println("WLAN verbunden");

  connectToStream();
}

// =========================================

void handleEncoder()
{
  int currentCLK = digitalRead(ENC_CLK);

  if (currentCLK != lastCLK)
  {
    if (digitalRead(ENC_DT) != currentCLK)
      volume++;
    else
      volume--;

    volume = constrain(volume, 0, 100);
    player.setVolume(volume);
  }

  lastCLK = currentCLK;
}

// =========================================

void handleStream()
{
  if (!client.connected())
    connectToStream();

  if (!client.available())
    return;

  if (!headerSkipped)
  {
    String line = client.readStringUntil('\n');

    if (line == "\r")
    {
      headerSkipped = true;
      Serial.println("Stream Start");
    }
  }
  else
  {
    if (digitalRead(VS1053_DREQ))
    {
      int bytesread = client.read(mp3buff, sizeof(mp3buff));

      if (bytesread > 0)
        player.playChunk(mp3buff, bytesread);
    }
  }
}

// =========================================

void loop()
{
  handleEncoder();
  handleStream();
  drawScreen();
}

  static int lastPrev=HIGH;

  int nextState=digitalRead(BTN_NEXT);
  int prevState=digitalRead(BTN_PREV);

  if(lastNext==HIGH && nextState==LOW)
  {
    station++;
    if(station>=stationCount) station=0;
    connectToStream();
  }

  if(lastPrev==HIGH && prevState==LOW)
  {
    station--;
    if(station<0) station=stationCount-1;
    connectToStream();
  }

  lastNext=nextState;
  lastPrev=prevState;
}


// STREAM
void handleStream()
{
  if(!client.connected())
    connectToStream();

  if(!client.available())
    return;

  if(!headerSkipped)
  {
    String line=client.readStringUntil('\n');

    if(line=="\r")
      headerSkipped=true;

    if(line.startsWith("StreamTitle"))
    {
      int start=line.indexOf("='");
      int end=line.indexOf("';");

      if(start>0 && end>start)
      {
        streamTitle=line.substring(start+2,end);
        scrollX=0;
      }
    }
  }
  else
  {
    if(digitalRead(VS1053_DREQ))
    {
      int bytesread=client.read(mp3buff,sizeof(mp3buff));

      if(bytesread>0)
        player.playChunk(mp3buff,bytesread);
    }
  }
}


// SCROLL TEXT
void handleScroll()
{
  if(millis()-lastScroll<80) return;

  int textWidth=u8g2.getStrWidth(streamTitle.c_str());

  if(textWidth>128)
  {
    scrollX++;

    if(scrollX>textWidth+20)
      scrollX=0;
  }

  lastScroll=millis();
}


// LOOP
void loop()
{
  handleEncoder();
  handleButtons();
  handleStream();
  handleScroll();

  if(millis()-lastDisplay>50)
  {
    drawScreen();
    lastDisplay=millis();
  }
}
